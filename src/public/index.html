function registerClient(req, res) { res.writeHead(200, { 'Content-Type': 'text/event-stream', 'Connection': 'keep-alive', 'Cache-Control': 'no-cache' }); res.write(': connected\n\n'); clients.push(res); req.on('close', () => { clients = clients.filter(c => c !== res); }); return res; } function broadcast(event, data) { const formattedMessage = `event: ${event}\ndata: ${JSON.stringify(data)}\n\n`; clients.forEach(res => res.write(formattedMessage)); } function getClientsCount() { return clients.length; } function clearClients() { clients = []; } module.exports = { registerClient, broadcast, getClientsCount, clearClients }; ``` - [ ] **Step 2: Créer le test unitaire tests/services/sseService.test.js** ```javascript const sseService = require('../../src/services/sseService'); const PriceHistory = require('../models/PriceHistory'); const sseService = require('./sseService'); let pollIntervalId = null; async function fetchBinancePrice() { try { const res = await fetch('https://api.binance.com/api/v3/ticker/price?symbol=BTCUSDT'); alert.triggeredAt = new Date(); await alert.save(); // Diffuser en direct via SSE sseService.broadcast('alertTriggered', alert); } } } catch (error) { console.error('Erreur lors de l\'évaluation des alertes:', error.message); Évaluer les alertes actives await checkAndTriggerAlerts(currentPrice); } function startPolling(intervalMs = 5000) { if (pollIntervalId) return; pollIntervalId = setInterval(processPriceCycle, intervalMs); // Premier appel immédiat processPriceCycle(); } function stopPolling() { if (pollIntervalId) { clearInterval(pollIntervalId); pollIntervalId = null; } } module.exports = { fetchBinancePrice, checkAndTriggerAlerts, processPriceCycle, startPolling, stopPolling };
